Smart Autonomous Monitoring & Control System

Project Objectives

The goal was to integrate multiple fundamental embedded systems concepts into one robust architecture.

Automation

Continuously monitor the environment and execute logic-based responses dynamically.

Safety Thresholds

Detect hazardous conditions such as gas leaks, extreme temperatures, or proximity breaches.

User Override

Provide physical interfaces allowing the user to seamlessly switch between operational states.

System Sub-Modules

The architecture is divided into six distinct operational requirements.

1

Basic Output Control

Discrete LEDs visually broadcast the overarching system state.

Green = SafeYellow = WarnRed = Alert

2

Variable Actuation

A PWM-driven LED linearly adjusts its brightness inversely proportional to ambient light (LDR).

3

State Management

Hardware interrupts/polling via tactile buttons to toggle Auto, Manual, and Emergency modes.

4

Audible Alerts

A piezoelectric buzzer is triggered with varying frequencies based on the severity of the hazard.

5

Telemetry Display

A 16x2 LCD renders real-time data including temperature, luminosity, distance, and current mode.

6

Environmental Sensing

An array of analog and digital sensors (LDR, DHT11, MQ-2, Ultrasonic) gather real-world data.

Operational Workflow

Detailed sequence of how the firmware evaluates variables and executes instructions.

Sensors capture analog/digital environmental data
→
Arduino evaluates variables against predefined safety thresholds
→
System actuates LEDs, PWM lights, buzzer, and updates LCD

Conditional Logic Framework

  • Lighting: If ambient light decreases, PWM duty cycle increases (brighter LED).
  • Hazard Detection: If gas levels spike, temperature exceeds safe bounds, or an obstacle breaches the minimum safe distance, emergency subroutines trigger.
  • Status: LED colors dynamically shift (Green ➔ Yellow ➔ Red) to reflect the current threat level.
  • Telemetry: The LCD screen polls hardware every 300ms to refresh the display.

Operational Modes

  • Auto Mode (Default): Full autonomous regulation utilizing continuous sensor polling.
  • Manual Mode: Fixed state with limited automated response; allows user stabilization.
  • Emergency Mode: Immediate override triggering maximum alert state (Red LED, high-pitch buzzer).

Source Code (Firmware)

The complete C++ implementation programmed onto the Arduino microcontroller.

main.ino — Developed by Khablil
#include <LiquidCrystal.h>
#include <DHT.h>

LiquidCrystal lcd(12, 11, 5, 4, 3, 2);

#define DHTPIN A0
#define DHTTYPE DHT11
DHT dht(DHTPIN, DHTTYPE);

const int ldrPin = A1;
const int gasPin = A2;
const int trigPin = 6;
const int echoPin = 7;

const int redLED = 8;
const int yellowLED = 9;
const int greenLED = 10;
const int pwmLED = 13;
const int buzzerPin = A3;

const int autoButton = A4;
const int manualButton = A5;
const int emergencyButton = A6;

int mode = 0;
float temperature = 0;
int lightValue = 0;
int gasValue = 0;
long duration = 0;
float distance = 0;

void setup() {
  lcd.begin(16, 2);
  lcd.print("Project by");
  lcd.setCursor(0, 1);
  lcd.print("Khablil");
  delay(2000);
  lcd.clear();

  dht.begin();
  pinMode(ldrPin, INPUT);
  pinMode(gasPin, INPUT);
  pinMode(trigPin, OUTPUT);
  pinMode(echoPin, INPUT);
  pinMode(redLED, OUTPUT);
  pinMode(yellowLED, OUTPUT);
  pinMode(greenLED, OUTPUT);
  pinMode(pwmLED, OUTPUT);
  pinMode(buzzerPin, OUTPUT);
  
  pinMode(autoButton, INPUT_PULLUP);
  pinMode(manualButton, INPUT_PULLUP);
  pinMode(emergencyButton, INPUT_PULLUP);
  
  Serial.begin(9600);
}

void loop() {
  readButtons();
  readSensors();

  if (mode == 2) {
    emergencyMode();
  } else if (mode == 1) {
    manualMode();
  } else {
    autoMode();
  }

  updateDisplay();
  delay(300);
}

void readButtons() {
  if (digitalRead(autoButton) == LOW) mode = 0;
  if (digitalRead(manualButton) == LOW) mode = 1;
  if (digitalRead(emergencyButton) == LOW) mode = 2;
}

void readSensors() {
  temperature = dht.readTemperature();
  if (isnan(temperature)) temperature = 0;

  lightValue = analogRead(ldrPin);
  gasValue = analogRead(gasPin);

  digitalWrite(trigPin, LOW);
  delayMicroseconds(2);
  digitalWrite(trigPin, HIGH);
  delayMicroseconds(10);
  digitalWrite(trigPin, LOW);

  duration = pulseIn(echoPin, HIGH);
  distance = duration * 0.034 / 2;
}

void autoMode() {
  int brightness = map(lightValue, 1023, 0, 0, 255);
  brightness = constrain(brightness, 0, 255);
  analogWrite(pwmLED, brightness);

  bool danger = false;
  bool warning = false;

  if (gasValue > 450 || temperature > 40 || distance < 10) {
    danger = true;
  } else if (gasValue > 300 || temperature > 30 || distance < 20) {
    warning = true;
  }

  if (danger) {
    digitalWrite(redLED, HIGH);
    digitalWrite(yellowLED, LOW);
    digitalWrite(greenLED, LOW);
    tone(buzzerPin, 1000);
  } else if (warning) {
    digitalWrite(redLED, LOW);
    digitalWrite(yellowLED, HIGH);
    digitalWrite(greenLED, LOW);
    tone(buzzerPin, 700);
    delay(150);
    noTone(buzzerPin);
  } else {
    digitalWrite(redLED, LOW);
    digitalWrite(yellowLED, LOW);
    digitalWrite(greenLED, HIGH);
    noTone(buzzerPin);
  }
}

void manualMode() {
  analogWrite(pwmLED, 150);
  digitalWrite(redLED, LOW);
  digitalWrite(yellowLED, HIGH);
  digitalWrite(greenLED, LOW);

  if (gasValue > 500 || temperature > 45 || distance < 8) {
    tone(buzzerPin, 1200);
  } else {
    noTone(buzzerPin);
  }
}

void emergencyMode() {
  analogWrite(pwmLED, 255);
  digitalWrite(redLED, HIGH);
  digitalWrite(yellowLED, LOW);
  digitalWrite(greenLED, LOW);
  tone(buzzerPin, 1500);
}

void updateDisplay() {
  lcd.clear();
  lcd.setCursor(0, 0);
  lcd.print("T:");
  lcd.print(temperature, 1);
  lcd.print("C ");
  lcd.print("L:");
  lcd.print(lightValue);

  lcd.setCursor(0, 1);
  if (mode == 0) lcd.print("AUTO ");
  else if (mode == 1) lcd.print("MANUAL ");
  else lcd.print("EMERG ");

  lcd.print("D:");
  lcd.print(distance, 0);
}
